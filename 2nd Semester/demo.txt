//P6: Function to print factorial of n using recursion
# include <stdio.h>
int factorial(int n);
int main() {
    int n;
    printf("enter n : ");
    scanf("%d", &n);
    printf("factorial is : %d", factorial(n));
    return 0;
}
int factorial(int n) {
    if(n == 0) {
        return 1;
    }
    int factnm1 = factorial(n-1);
    int factn = factnm1 * n;
    return factn;
}


//P7: Function to print fibonacci series using recursion
#include<stdio.h>
int fibonacci(int num){
    if (num == 0){
        return 0;
    }
    else if (num == 1){
        return 1; 
    }
    else{
        return fibonacci(num - 1) + fibonacci(num - 2);
    }
}
int main()
{
    int num; 
    printf("Enter the number of elements to be in the series : ");
    scanf("%d", &num); 

    for (int i = 0; i < num; i++){
        printf("%d ", fibonacci(i)); 
    }
    return 0;
}

//P8: Program to swap 2 integers
# include <stdio.h>
void swap(int a, int b);
void _swap(int* a, int *b);
int main() {
    int x = 3, y = 5;
    
    //call by value
    swap(x, y);
    printf("x = %d & y = %d\n", x, y);
    
    //call by reference
    _swap(&x, &y);
    printf("x = %d & y = %d\n", x, y);
    
    return 0;
}
void swap(int a, int b) {
    int t = a;
    a = b;
    b = t;
}
void _swap(int* a, int* b) {
    int t = *a;
    *a = *b;
    *b = t;
}

    printf("n is : %d\n", number);
    //call by reference

    return 0;
}
void square(int n) {
    n = n * n;
    printf("square is : %d\n", n);
}
void _square(int* n) {
    *n = *n * *n;
    printf("square is : %d\n", *n);
}

int countOdd (int arr[], int n) {
    int count = 0;
    for(int i=0; i<n; i++) {
        if(arr[i] % 2 != 0) { // odd
            count++;
        }
    }
    return count;
}

//P24: Reverse an array
void reverse(int arr[], int n);
void printArr(int arr[], int n);
int main() {
    int arr[] = {1, 2, 3, 4, 5};
    reverse(arr, 5);
    printArr(arr, 5);
    return 0;
}
void printArr(int arr[], int n) {
    for(int i=0; i<n; i++) {
        printf("%d\t", arr[i]);
    }
    printf("\n");
}
void reverse(int arr[], int n) {
    for(int i=0; i<n/2; i++) {
        int firstVal = arr[i];
        int secondVal = arr[n-i-1];
        arr[i] = secondVal;
        arr[n-i-1] = firstVal;
    }
}
